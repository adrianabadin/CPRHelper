5. Add ExportPdfCommand (after BackToList command):
   ```csharp
   [RelayCommand(CanExecute = nameof(CanExport))]
   private async Task ExportPdf()
   {
       if (SelectedSession is null) return;
       IsExporting = true;
       UpdateExportCommands();
       try
       {
           var events = SessionEvents.ToList();
           var filePath = await _pdfExportService.GeneratePdfAsync(SelectedSession, events);
           
           // Save local copy to app data
           var localDir = Path.Combine(FileSystem.Current.AppDataDirectory, "Exports");
           Directory.CreateDirectory(localDir);
           var localPath = Path.Combine(localDir, Path.GetFileName(filePath));
           File.Copy(filePath, localPath, overwrite: true);
           
           // Open share sheet
           await Share.RequestAsync(new ShareFileRequest
           {
               Title = "Exportar PDF — ACLS Tracker",
               File = new ShareFile(filePath),
               PresentationSourceBounds = DeviceInfo.Platform == DevicePlatform.iOS
                   ? new Rect(0, 0, 0, 0) : Rect.Zero
           });
       }
       catch (Exception ex)
       {
           await Shell.Current.DisplayAlert("Error", $"No se pudo generar el PDF: {ex.Message}", "OK");
       }
       finally
       {
           IsExporting = false;
           UpdateExportCommands();
       }
   }
   ```

IMPORTANT: The `IsExporting` property drives CanExecute on both commands AND disables buttons via IsEnabled binding. The CommunityToolkit.Mvvm source generator creates the command properties from `[RelayCommand]`. Since we use `CanExecute`, the buttons will auto-disable during export. If InvertedBoolConverter doesn't exist in the project, use the CanExecute mechanism only (the `[RelayCommand(CanExecute = ...)]` pattern handles this natively — the Button becomes disabled when CanExport returns false). In that case, remove the `IsEnabled` binding and rely solely on the command's CanExecute.
